module _ (Adv : Participant) (Adv∉ : Adv ∉ Hon) where
  open AdvM Adv Adv∉

  MaximalRun : Strategies → CRun → Set
  MaximalRun SS R = (R -conforms-to- SS)
                  × (¬ ∃ λ R′ → (R′ -conforms-to- SS)
                              × (length (toList R′) > length (toList R)))

  unique-maximal-run : ∀ {R R′} →
    ∙ MaximalRun SS R
    ∙ MaximalRun SS R′
      ────────────────
      R ≡ R′
